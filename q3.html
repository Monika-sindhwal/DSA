        #include<stdio.h>
        struct queue{
            int data;
            struct queue *next;
        };
        struct queue *rear=NULL,*front=NULL,*nn,*temp;
        void print(){
            temp=front;
            while(temp->next!=NULL){
            printf("%d\t",temp->data);
            temp=temp->next;}
        }
        void isfull(){
            nn=(struct queue*)malloc(sizeof(struct queue));
            if(nn=NULL)
            printf("Queue is full");
            else
            printf("Queue is not full");
        }
        void isEmpty(){
            if(front==NULL)
            printf("Queue is empty");
            else
            printf("Queue is not empty");
        }
        void insert(int value){
            nn=(struct queue*)malloc(sizeof(struct queue));
            if(nn==NULL)
            printf("Overflow condition");
            else{
                 if(rear==NULL){
                rear=nn;
                front=nn;
                nn->data=value;
                nn->next=NULL;
            }
            else{
                nn->data=value;
                nn->next=NULL;
                rear->next=nn;
                rear=nn;
            }
            }
           
        }
        void delete(){
            if(front==NULL)
            printf("Underflow condition");
            else if(front==rear){
                printf("%d is popped\n",front->data);
                front=NULL;
            }
            else{
                printf("%d is popped\n",front->data);
                front=front->next;  
            }
        }
        void peek(){
            if(rear==NULL)
            printf("Queue is empty\n");
            else
            printf("%d\n",rear->data);
        }
