        #include <stdio.h>
        #include <string.h>
            
        void selectionSort(char arr[][20], int n) {
                int i, j, minIndex;
                char temp[20];
            
                for (i = 0; i < n-1; i++) {
                    minIndex = i;
                    for (j = i+1; j < n; j++) {
                        if (strcmp(arr[j], arr[minIndex]) < 0) {
                            minIndex = j;
                        }
                    }
                    if (minIndex != i) {
                        strcpy(temp, arr[i]);
                        strcpy(arr[i], arr[minIndex]);
                        strcpy(arr[minIndex], temp);
                    }
                }
            }
            
            int main() {
                int n=5, i;
            
                char str[5][20];
            
                for(i = 0; i < n; i++) {
                    printf("Enter string %d: ", i+1);
                    gets(str[i]);
                }
            
                
                selectionSort(str, n);
            
               
                printf("\nSorted array:\n");
                for(i = 0; i < n; i++) {
                    printf("%s\n", str[i]);
                }
            
                return 0;
            }
