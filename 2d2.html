            void add(int sum[2][2], int r, int c, int arr[2][2], int arr2[2][2])
            {
                int i, j;
            
                for (i = 0; i < r; i++)
                {
                    for (j = 0; j < c; j++)
                    {
                        sum[i][j] = arr[i][j] + arr2[i][j];
                    }
                }
            }
            int main()
            {
                int arr[2][2], i, j, arr2[2][2], sum[2][2];
                int row = 2, col = 2;
                printf("Enter the first matrix:\n");

                printf("Enter the second matrix:\n");
                assign(arr2, row, col);
                add(sum, row, col, arr, arr2);

                {
