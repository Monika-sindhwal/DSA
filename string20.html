        #include<stdio.h>
            int countcharacters(char *string){
                int count=0;
                for(int i=0;string[i]!='\0';i++){
                    count++;
                }
                return count;
            }
            int countwords(char *string){
                int count=0;
                for(int i=0;string[i]!='\0';i++){
                    if(string[i]==32 ){
                        count++;
                    }
                }
                return count+1;
            }
            int countspaces(char *string){
                int count=0;
                for(int i=0;string[i]!='\0';i++){
                    if(string[i]==32 ){
                        count++;
                    }
                }
                return count;
            }
            int countlines(char *string){
                int count=0,i=0;
                while(string[i]!='\0'){
                    if(string[i]=='.')
                    count++;
                    i++;
                }
                return count;
            }
            int main()
            {
            char string[20];
            int character,words,spaces,lines;
            gets(string);
            character=countcharacters(string);
            printf("No. of characters are %d\n",character);
            words=countwords(string);
            printf("No. of words are %d\n",words);
            spaces=countspaces(string);
            printf("No. of spaces are %d\n",spaces);
            lines=countlines(string);
            printf("No. of lines are %d\n",lines);
            
            return 0;
            }
