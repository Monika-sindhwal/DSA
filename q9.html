#include<stdio.h>
//Double ended queue
int rear=-1,front=-1;
void print(int *arr){
    int i;
    for(i=front;i<=rear;i++){
        printf("\n%d\t",arr[i]);
    }
}
void insertatbeg(int arr[]){
    int value;
if(rear==-1 && front==-1){
    printf("Enter the value to be inserted:");
    scanf("%d",&value);
    rear++;
    front++;
    arr[rear]=value;
}
else if(front==0){
printf("Queue is full from the beginning\n");
}
else{
     printf("Enter the value to be inserted:");
    scanf("%d",&value);
     front--;
    arr[front]=value;
   
}

}
void peek(int arr[]){
     if(front==-1&&rear==-1){
        printf("Queue is empty\n");
    }
    else
    printf("%d",arr[rear]);
}
void deleteatlast(int arr[]){
    if((front==-1 && rear==-1)||front>rear)
    printf("queue is empty\n");
    else if(rear==0){
        printf("%d is deleted\n",arr[rear]);
        front=-1;
        rear=-1;
    }
    else{
        printf("%d is deleted\n",arr[rear]);
        rear=rear-1;
    }
}
void deletefromfront(int arr[]){
    if((front==-1&&rear==-1)||front>rear)
    printf("queue is empty\n");
    else{
    printf("%d is deleted\n",arr[front]);
    front++;
    }
}
void insertfromrear(int arr[],int size){
    int value;
    printf("size=%d\n",size);
    
     if(rear==size-1){
printf("Queue is full\n");
}
else if(rear==-1 && front==-1){
    front=0;
    rear=0;
    printf("Enter the value to be inserted:");
    scanf("%d",&value);
    arr[rear]=value;
}

else{
    printf("Enter the value to be inserted:");
    scanf("%d",&value);
    rear=rear+1;
     printf("rear=%d\n",rear);

    arr[rear]=value;
}
}
void isEmpty(int arr[]){
    if(front==-1&&rear==-1){
        printf("Queue is empty\n");
    }
    else
    printf("Queue is not empty\n");

}
void isFull(int *arr,int size){
if(front==-1 && rear==size-1){
printf("Queue is full\n");
}
else{
    printf("Queue is not full");
}
}
int main()
{
    int arr[5],ch;
    while(1){
    printf("Choose 1 option:\n1.Insert at beginning\n2.Insert at end\n3.Delete at beginnig\n4.Delete at end\n5.Peek\n6.Isempty\n7.Isfull\n8.exit\n9.Print\n");
    scanf("%d",&ch);
    switch(ch){

return 0;
