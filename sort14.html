            void sort(char arr[n][20], int i, int j) {
                if (i < n - 1) {
                    j = i + 1;
                    swap(arr, j, i);
                    i = i + 1;
                    return sort(arr, i, j);
                }
                else
                    return;
            }
            
            void swap(char arr[n][20], int j, int i) {
                char temp[20];
                if (j < n - i) {

                        strcpy(arr[j + 1], temp);

                    j = j + 1;
