            struct tree{
                int data;
                struct tree *left;
                struct tree *right;
            };
            struct tree *start,*nn,*search,*prev,*temp;
            
            
            
            //Code for insertion
            
            void insert(){
                int value;
                printf("Enter the value of node which you want to insert:");
                scanf("%d",&value);
                if(start==NULL){
                   start=(struct tree*)malloc(sizeof(struct tree));
                    start->data=value;
                    start->left=start->right=NULL;
                }
               else{
                   nn=(struct tree*)malloc(sizeof(struct tree));
                   nn->data=value;
                   nn->left=nn->right=NULL;
                   prev=NULL;
                   search=start;
                   
                   while(search!=NULL){
                  if(search->data>value){
                      prev=search;
                      search=search->left;
                  }
                 else if(search->data<value){
                     prev=search;
                     search=search->right;
                 }
               }
                  if(prev->data>value){
                  printf("inserted at left\n");
                  prev->left=nn;}
                  if(prev->data<value){
                  printf("inserted at right\n");
                  prev->right=nn;
               }
                   
               }
            }
            //       Inorder traversal
            
            int inTraverse(struct tree *root) {
                static int count=0;
             
              if (root != NULL) 
              {
                inTraverse(root -> left); 
                count++;
                inTraverse(root -> right);  
              }
              return count;
            }
