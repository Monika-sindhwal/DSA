        #include <stdio.h>
            int fact(int num)
            {
                if (num == 1)
                    return 1;
                else
                    return num * fact(num - 1);
            }
            int main()
            {
                int num;
                printf("Enter the no. whose factorial you want to find:");
                scanf("%d", &num);
                printf("Factorial of %d is %d", num, fact(num));
                return 0;
            }
