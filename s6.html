        #include <stdio.h>
            int bs(int l, int r, char key, char *arr)
            {
                if (l <= r)
                {
                    int mid;
                    mid = ((l + r) / 2);
                    if (arr[mid] == key)
                    {
                        return mid;
                    }
                    else if (arr[mid] > key)
                        return bs(l, mid - 1, key, arr);
                    else
                        return bs(mid + 1, r, key, arr);
                }
                return -1;
            }
