        #include <stdio.h>
            void assign(int arr[4][2], int r, int c)
            {
                int i, j;
            
                for (i = 0; i < r; i++)
                {
                    printf("Enter the elements of %d row:", i + 1);
                    for (j = 0; j < c; j++)
                    {
                        scanf("%d", &(arr[i][j]));
                    }
                }
            }
            int main()
            {
                int arr[4][2], i, j;
                int rows = 4, cols = 2;
                assign(arr, rows, cols);
                printf("The matrix is:\n");
                for (i = 0; i < rows; i++)
                {
                    for (j = 0; j < cols; j++)
                    {
                        printf("%d\t", arr[i][j]);
                    }
                    printf("\n");
                }
                return 0;
            }
