        #include 
            float mean(int arr[], int n)
            {
                if (n == 1)
                    return arr[n - 1];
                else
                    /*multiplying by n-1 as we need sum of left integers only if we will take average of previous no. and then take average with final no. it will give wrong answer*/
            
                    return (arr[n - 1] + ((n - 1) * mean(arr, n - 1))) / n;
            }
            int main()
            {
                int arr[5], i, n = 5;
                printf("Enter 5 elements of the array\n");
                for (i = 0; i < n; i++)
                {
                    printf("Enter %d element:", i + 1);
                    scanf("%d", &arr[i]);
                }
                printf("The arithmetic mean of given integer array is:%f", mean(arr, n));
                return 0;
            }
