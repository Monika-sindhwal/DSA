            struct stack{
                int value;
                struct stack *next;
            };
            struct stack *start,*nn,*t;
            void push(int value){
                if(start==NULL){
                    start=(struct stack*)malloc(sizeof(struct stack));
                    start->value=value;
                    start->next=NULL;
                      
                }
                else{
                    nn=(struct stack*)malloc(sizeof(struct stack));
                    nn->value=value;
                    nn->next=start;
                    start=nn;
                   
                }
            }
            void pop(){
                if(start==NULL)
                printf("Underflow condition\n");
                else{
                t=start;
                start=start->next;
                printf("%d",t->value);
                free(t);
                }
            
            }\
