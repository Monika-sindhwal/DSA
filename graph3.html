          else
        ver[s][d]=1;
    }}
        
}

void push(int vertex) {
    stack[++top] = vertex;
}

int pop() {
    return stack[top--];
}


void dfsAdjacencyMatrix(int startVertex) {
    push(startVertex);
    visited[startVertex] = 1;

    while (top!=-1) {
        int currentVertex = pop();
        printf("%d ", currentVertex);

        for (int i = 0; i < v; i++) {
            if (ver[currentVertex][i] == 1 && !visited[i]) {
                push(i);
                visited[i] = 1;
            }
        }
    }
}
