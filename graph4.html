#include <stdio.h>
#include<stdlib.h>
struct edge{
    int data;
    struct edge *next2;
};
int top=-1,stack[10],visited[10];
void push(int value){
    top++;
    stack[top]=value;
}
int pop(){
    return stack[top--];
}
struct vertix{
    int data;
    struct vertix *next;
    struct edge *ptr;
};
struct vertix *start=NULL,*search,*nn,*t,*pre;
struct edge *s,*nn2,*pre2;
void insertvertex(){
    if(start==NULL){
        start=(struct vertix*)malloc(sizeof(struct vertix));
        printf("Enter the vertex:");
        scanf("%d",&(start->data));
        start->next=NULL;
        start->ptr=NULL;
    }
    else{
      search=start;
      nn=(struct vertix*)malloc(sizeof(struct vertix));
      while(search->next!=NULL){
          search=search->next;
      }
      search->next=nn;
      printf("Enter the vertex:");
      scanf("%d",&(nn->data));
      nn->next=NULL;
      nn->ptr=NULL;
      
    }
}
void insertedge(){
    int src,dest;
    printf("Enter the source and destination:");
    scanf("%d %d",&src ,&dest);
    search=start;
    while(search!=NULL){
        if(search->data==src)
        break;
        search=search->next;
    }
    if(search==NULL){
        printf("Entered source is not available");
        return;
    }
     t=start;
    while(t!=NULL){
        if(t->data==dest)
        break;
        t=t->next;
    }
    if(t==NULL){
        printf("Entered dest is not available");
        return;
    }
    nn2=(struct edge*)malloc(sizeof(struct edge));
    nn2->data=dest;
    nn2->next2=NULL;
    if(search->ptr==NULL){
    search->ptr=nn2;
   
    }
    else{
       s=search->ptr;
       while(s->next2!=NULL){
           s=s->next2;
       }
       s->next2=nn2;
       
    }
   
    
}
void dfs(int sv){
    push(sv);
    visited[sv]=1;
    while(top!=-1){
         int currentVertex = pop();
         printf("%d ", currentVertex);
         search=start;
         while(search!=NULL){
             if(search->data==currentVertex){
                 break;
             }
             search = search->next;
         } if (search != NULL) {
             s=search->ptr;
             while(s!=NULL){
                 if(!visited[s->data]){
                     push(s->data);
                     visited[s->data]=1;
                 }
                 s=s->next2;
             }
         }
    }
}
