// BST through array
#include <stdio.h>
#include <stdbool.h>

#define MAX_SIZE 50

int bst[MAX_SIZE]; 

int root = -1;

void insert(int key)
{
    if (root == -1)
    {
        root = 0;
        bst[root] = key;
    }
    else
    {
        int current = root;
        while (true)
        {
            if (key < bst[current])
            {
                if (bst[current * 2 + 1] == 0)
                {
                    bst[current * 2 + 1] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 1;
                }
            }
            else
            {
                if (bst[current * 2 + 2] == 0)
                {
                    bst[current * 2 + 2] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 2;
                }
            }
        }
    }
}
void deletenode(int val)
{
    if (MAX_SIZE == 0)
    {
        printf("Tree is Empty!");
    }

    int i = 0;
    // getting correct index.
    while (i < MAX_SIZE && bst[i] != -1)
    {
        if (bst[i] == val)
        {
            break;
        }
        if (val < bst[i])
        {
            i = 2 * i + 1;
        }
        if (val > bst[i])
        {
            i = 2 * i + 2;
        }
    }
    if (i < MAX_SIZE && bst[i] != -1)
    {
        int val = bst[i];
        bst[i] = -1;
        printf("Value %d is deleted\n ", val);
    }
    else
    {
        printf("Value not found!");
    }
}
void search(int key)
{
    int current = root;
    while (bst[current] != 0)
    {
        if (key == bst[current])
        {
            printf("Value found\n");
            return;
        }
        else if (key < bst[current])
        {
            current = current * 2 + 1;
        }
        else
        {
            current = current * 2 + 2;
        }
    }
    if (bst[current] == 0)
    {
        printf("Value not found\n");
    }
}

void inorder(int node)
{
    if (node >= MAX_SIZE || bst[node] == 0)
    {
        return;
    }

    inorder(node * 2 + 1);
    printf("%d ", bst[node]);
    inorder(node * 2 + 2);
}

void preorder(int node)
{
    if (node >= MAX_SIZE || bst[node] == 0)
    {
        return;
    }
    printf("%d ", bst[node]);
    preorder(node * 2 + 1);
    preorder(node * 2 + 2);
}

void postorder(int node)
{
    if (node >= MAX_SIZE || bst[node] == 0)
    {
        return;
    }
    postorder(node * 2 + 1);
    postorder(node * 2 + 2);
    printf("%d ", bst[node]);
}

int main()
{
    int op, value;
    for (int i = 0; i < 15; i++)
    {
        bst[i] = 0;
    }
    while (1)
    {
        printf("Enter 1 option:\n1.Insert\n2.Delete\n3.Search\n4.inorder\n5.exit\n6.Postorder\n7.Preorder\n");
        scanf("%d", &op);
        switch (op)
        {
        case 1:
            printf("Enter the value to be inserted:");
            scanf("%d", &value);
            insert(value);
            break;
        case 2:
            printf("Enter the value to be deleted:");
            scanf("%d", &value);
            deletenode(value);
            break;
        case 3:

            printf("Enter the value to be searched:");
            scanf("%d", &value);
            search(value);
            break;
        case 4:
            inorder(root);
            break;
        case 5:
            exit(0);
            break;
        case 6:
            postorder(root);
            break;
        case 7:
            preorder(root);
            break;
        default:
            printf("Invalid input!");
        }
    }
    return 0;
}
