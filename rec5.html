        #include <stdio.h>
            int max(int *arr, int n)
            {
                int maximum;
                if (n == 1)
                    return arr[n - 1];
                else
                {
                    maximum = max(arr, n - 1);
                    if (maximum > arr[n - 1])
                        return maximum;
                    else
                        return arr[n - 1];
                }
            }
            int main()
            {
                int arr[5], i, n = 5;
                printf("Enter 5 elements of the array\n");
                for (i = 0; i < 5; i++)
                {
                    printf("Enter %d element:", i + 1);
                    scanf("%d", &arr[i]);
                }
                printf("The largest element in the entered array is %d", max(arr, n));
                return 0;
            }
