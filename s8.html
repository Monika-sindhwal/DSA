        #include <stdio.h>
            int bs(char arr[], int size, char key)
            {
                int l = 0, u = size - 1, mid;
                //=u because for end most elements like 0 or last element which are not mid 0,0 comesor 5,5
                while (l <= u)
                {
                    mid = (l + u) / 2;
                    if (arr[mid] > key)
                        u = mid - 1;
                    else if (arr[mid] < key)
                    {
                        l = mid + 1;
                    }
                    else
                        return mid;
                }
                // or if(l>u)
                return -1;
            }
            int main()
            {
                char arr[10], key;
                int loc;
                printf("Enter the sorted string:");
                gets(arr);
                printf("Enter the element you want to search:");
                scanf("%c", &key);
                loc = bs(arr, 10, key);
                if (loc == -1)
                    printf("%c not found in the array", key);
                else
                    printf("%c fount at index %d in the array", key, loc);
                return 0;
            }
