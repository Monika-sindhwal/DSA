        #include <stdio.h>
            int bs(int arr[], int size, int key)
            {
                int l = 0, u = size - 1, mid;
                //=u because for end most elements like 0 or last element which are not mid 0,0 comesor 5,5
                while (l <= u)
                {
                    mid = (l + u) / 2;
                    if (arr[mid] > key)
                        u = mid - 1;
                    else if (arr[mid] < key)
                    {
                        l = mid + 1;
                    }
                    else
                        return mid;
                }
                // or if(l>u)
                return -1;
            }
            int main()
            {
                int arr[5], key, loc;
                printf("Enter 5 elements of the array:");
                for (int i = 0; i < 5; i++)
                    scanf("%d", &arr[i]);
                printf("Enter the element you want to search:");
                scanf("%d", &key);
                loc = bs(arr, 5, key);
                if (loc == -1)
                    printf("%d not found in the array", key);
                else
                    printf("%d fount at index %d in the array", key, loc);
                return 0;
            }
