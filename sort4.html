    #include<stdio.h>

        void bubbleSort(int arr[], int size) {
            if (size == 1)
                return;
        
            for (int i = 0; i < size - 1; i++) {
                if (arr[i] > arr[i + 1]) {
                    // Swap arr[i] and arr[i+1]
                    int temp = arr[i];
                    arr[i] = arr[i + 1];
                    arr[i + 1] = temp;
                }
            }
        
            
            bubbleSort(arr, size - 1);
        }
        
        int main() {
            int arr[5];
            printf("Enter 5 elements of the array:");
            for (int i = 0; i < 5; i++)
                scanf("%d", &arr[i]);
            int size = 5;
            printf("Original array: ");
            for (int i = 0; i < size; i++) {
                printf("%d ", arr[i]);
            }
        
            bubbleSort(arr, size);
        
            printf("\nSorted array: ");
            for (int i = 0; i < size; i++) {
                printf("%d ", arr[i]);
            }
        
            return 0;
        }
