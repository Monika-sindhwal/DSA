    #include<stdio.h>

        void selectionSort(int arr[], int size, int startIndex)
        {
            if (startIndex == size - 1)
                return;
        
            int min = startIndex;
            for (int j = startIndex + 1; j < size; j++)
            {
                if (arr[j] < arr[min])
                    min = j;
            }
        
            if (min != startIndex)
            {
                int temp = arr[startIndex];
                arr[startIndex] = arr[min];
                arr[min] = temp;
            }
        
            selectionSort(arr, size, startIndex + 1);
        }
        
        int main()
        {
            int arr[5];
            printf("Enter 5 elements of the array:");
            for (int i = 0; i < 5; i++)
                scanf("%d", &arr[i]);
            int size = 5;
        
            printf("Original array: ");
            for (int i = 0; i < size; i++)
            {
                printf("%d ", arr[i]);
            }
        
            selectionSort(arr, size, 0);
        
            printf("\nSorted array: ");
            for (int i = 0; i < size; i++)
            {
                printf("%d ", arr[i]);
            }
        
            return 0;
        }
