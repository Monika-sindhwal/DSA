        int insertion(int arr[],int i)
        {   if(i>=5)
             return;
           else{
             int curr=arr[i];
               int pre=i-1;
            if(curr<arr[pre])
            {  pre=swap(arr,curr,pre);
            
                arr[pre+1]=curr;
            }
            insertion(arr,i+1);
        }}
        
        int swap(int *arr,int curr,int pre)
        
        { if(pre>=0 && curr<arr[pre])
           {   
           arr[pre+1]=arr[pre];
           pre--;
           return swap(arr,curr,pre);
           }
           else
            return pre;
           
        }
