        #include <stdio.h>

            int main() {
                char str[30];
                char ch;
            
                printf("Enter a string: ");
                gets(str);
            
                printf("Enter a character to search: ");
                scanf("%c", &ch);
            
                int index = -1;
            
                for (int i = 0; str[i] != '\0'; i++) {
                    if (str[i] == ch) {
                        index=i;
                        break;
                    }
                }
            
                if (index!=-1) {
                    printf("The character '%c' was found in the string at index %d.\n",ch,index);
                } else {
                    printf("The character '%c' was not found in the string.\n", ch);
                }
            
                return 0;
            }
