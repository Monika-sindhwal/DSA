        #include <stdio.h>

            int main() {
                char str[30];
            
                printf("Enter a string: ");
                gets(str);
            
                
                for (int i = 0; str[i] != '\0'; i++) {
                    if (str[i + 1] != '\0') {
                        int difference = str[i + 1] - str[i];
                        printf("Difference between '%c' and '%c' is %d\n", str[i], str[i + 1], difference);
                    }
                }
            
                return 0;
            }
