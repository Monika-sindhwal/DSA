        #include <stdio.h>
            int min(int *arr, int n)
            {
                int minimum;
                if (n == 1)
                    return arr[n - 1];
                else
                {
                    minimum = min(arr, n - 1);
                    if (minimum > arr[n - 1])
                        return arr[n - 1];
                    else
                        return minimum;
                }
            }
            int main()
            {
                int arr[5], i, n = 5;
                printf("Enter 5 elements of the array\n");
                for (i = 0; i < 5; i++)
                {
                    printf("Enter %d element:", i + 1);
                    scanf("%d", &arr[i]);
                }
                printf("The smallest element in the entered array is %d", min(arr, n));
                return 0;
                return 0;
            }
