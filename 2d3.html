            void subtract(int sum[2][2],int r,int c,int arr[2][2],int arr2[2][2]){
                int i,j;
               
                for(i=0;i<r;i++){
                 for( j=0;j<c;j++){
                 sum[i][j]=arr[i][j]-arr2[i][j];
                 } 
                }
            }
