#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100

int bst[MAX_SIZE]; 
int root = -1; 

void insert(int key)
{
    if (root == -1)
    {
        root = 0;
        bst[root] = key;
    }
    else
    {
        int current = root;
        while (1)
        {
            if (key < bst[current])
            {
                if (bst[current * 2 + 1] == 0)
                {
                    bst[current * 2 + 1] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 1;
                }
            }
            else
            {
                if (bst[current * 2 + 2] == 0)
                {
                    bst[current * 2 + 2] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 2;
                }
            }
        }
    }
}

void inorder(int node)
{
    if (node >= MAX_SIZE || bst[node] == 0)
    {
        return;
    }

    inorder(node * 2 + 1);
    printf("%d ", bst[node]);
    inorder(node * 2 + 2);
}
