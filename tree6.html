        #include<stdio.h>
        #include<stdlib.h>
        struct tree{
            int data;
            struct tree *left;
            struct tree *right;
        };
        struct tree *start,*nn,*search,*prev,*temp;
        
        //Depth of tree
        
        int height(struct tree* root){
            if(root==NULL)
               return 0;
            else{
            int lheight=height(root->left);
           int rheight=height(root->right);
             
             if(lheight>rheight)
               return lheight+1;
             else
               return rheight+1;
            }
        }
        
        //Code for insertion
        
        void insert(){
            int value;
            printf("Enter the value of node which you want to insert:");
            scanf("%d",&value);
            if(start==NULL){
               start=(struct tree*)malloc(sizeof(struct tree));
                start->data=value;
                start->left=start->right=NULL;
            }
           else{
               nn=(struct tree*)malloc(sizeof(struct tree));
               nn->data=value;
               nn->left=nn->right=NULL;
               prev=NULL;
               search=start;
               
               while(search!=NULL){
              if(search->data>value){
                  prev=search;
                  search=search->left;
              }
             else if(search->data<value){
                 prev=search;
                 search=search->right;
             }
           }
              if(prev->data>value){
              printf("inserted at left\n");
              prev->left=nn;}
              if(prev->data<value){
              printf("inserted at right\n");
              prev->right=nn;
           }
               
           }
        }
        //       Inorder traversal
        
        void inTraverse(struct tree *root) {
          if (root != NULL) 
          {
            inTraverse(root -> left); 
            printf(" %d ", root -> data);
            inTraverse(root -> right);  
          }
        }
        
        int main()
        {
        int op,value,h;
        while(1){
            printf("\nEnter 1 option:\n1.Insert\n2.Depth of tree\n3.Exit\n4. Inorder traversal\n");
            scanf("%d",&op);
            
                if(op==1) 
                  insert();
                else if(op==2) {
                  h= height(start);
              printf("Levels are= %d",h);
                }
               else if(op==3)
                  break;
               
                else if(op==4)
               inTraverse(start);
               
               else
               printf("Invalid option!");
            
        }
        return 0;
        }
