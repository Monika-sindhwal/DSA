        #include <stdio.h>
            void assign(int arr[2][2], int r, int c)
            {
                int i, j;
            
                for (i = 0; i < r; i++)
                {
                    printf("Enter the elements of %d row:\n", i + 1);
                    for (j = 0; j < c; j++)
                    {
                        scanf("%d", &(arr[i][j]));
                    }
                }
            }
            void scalar(int arr[2][2], int r, int c, int sc)
            {
                int i, j;
                for (i = 0; i < r; i++)
                {
                    for (j = 0; j < c; j++)
                    {
                        arr[i][j] = sc * arr[i][j];
                    }
                }
            }
            int main()
            {
                int arr[2][2], i, j;
                int rows = 2, cols = 2, sc;
                printf("Enter the scalar quantity you want to multiply the matrix from:");
                scanf("%d", &sc);
                assign(arr, rows, cols);
                scalar(arr, rows, cols, sc);
                printf("The matrix is:\n");
                for (i = 0; i < rows; i++)
                {
                    for (j = 0; j < cols; j++)
                    {
                        printf("%d\t", arr[i][j]);
                    }
                    printf("\n");
                }
                return 0;
            }
