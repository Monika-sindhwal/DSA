        struct queue
{
    char data[30];
    struct queue *next;
};
struct queue *rear = NULL, *front = NULL, *nn, *temp;

void isfull()
{
    nn = (struct queue *)malloc(sizeof(struct queue));
    if (nn == NULL)
        printf("Queue is full");
    else
        printf("Queue is not full");
}
void isEmpty()
{
    if (front == NULL)
        printf("Queue is empty");
    else
        printf("Queue is not empty");
}
void insert()
{
    nn = (struct queue *)malloc(sizeof(struct queue));
    if (nn == NULL)
        printf("Overflow condition");
    else
    {
        if (rear == NULL)
        {
            rear = nn;
            front = nn;
            printf("Enter the string:");
            scanf("%s", &nn->data);
            nn->next = NULL;
        }
        else
        {
            printf("Enter the string:");
            scanf("%s", nn->data);
            nn->next = NULL;
            rear->next = nn;
            rear = nn;
        }
    }
}
void delete()
{
    if (front == NULL)
        printf("Underflow condition\n");
    else if (front == rear)
    {
        printf("%s is popped\n", front->data);
        front = NULL;
        rear = NULL;
    }
    else
    {
        printf("%s is popped\n", front->data);
        front = front->next;
    }
}
void peek()
{
    if (rear == NULL)
        printf("Queue is empty\n");
    else
        printf("%s\n", rear->data);
}
