#include <stdio.h>
#include <stdbool.h>

#define MAX_SIZE 100

int bst[MAX_SIZE]; // Define the array to store the BST

int root = -1; // Initialize the root of the tree

void insert(int key)
{
    if (root == -1)
    {
        root = 0;
        bst[root] = key;
    }
    else
    {
        int current = root;
        while (true)
        {
            if (key < bst[current])
            {
                if (bst[current * 2 + 1] == 0)
                {
                    bst[current * 2 + 1] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 1;
                }
            }
            else
            {
                if (bst[current * 2 + 2] == 0)
                {
                    bst[current * 2 + 2] = key;
                    break;
                }
                else
                {
                    current = current * 2 + 2;
                }
            }
        }
    }
}

void inorder(int node)
{
    if (node >= MAX_SIZE || bst[node] == 0)
    {
        return;
    }

    inorder(node * 2 + 1);
    printf("%d ", bst[node]);
    inorder(node * 2 + 2);
}
