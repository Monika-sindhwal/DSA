        int dequeue() {
            if (size == 0) {
                printf("Queue is Empty!");
                return -1;
            }
            if (size == 1) {
                int val = queue[front];
                front = rear = -1;
                size = 0;
                return val;
            }
            int val = queue[front];
            for (int i = front + 1; i <= rear; i++) {
                queue[i - 1] = queue[i];
            }
            rear--;
            size--;
            return val;
        }
        
        void display() {
            if (size == 0) {

                return;

            printf("\n");
        }

        void reverseQueue(int size) {
            int i = 0;
            int arr[size];
            while (i < size) {
                arr[i] = dequeue();
                i++;
            }
            for (int j = size - 1; j >= 0; j--) {
                enqueue(arr[j]);
            }
        }
