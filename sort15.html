#include <stdio.h>
#include <string.h>
void insertionSort(char arr[][20], int n) {
    int i, j;
    char key[20];

    for (i = 1; i < n; i++) {
        strcpy(key, arr[i]);
        j = i - 1;

        while (j >= 0 && strcmp(arr[j], key) > 0) {
            strcpy(arr[j + 1], arr[j]);
            j = j - 1;
        }
        strcpy(arr[j + 1], key);
    }
}

int main() {
    int n=5, i;
    char strArr[n][20];
    for(i = 0; i < n; i++) {
        printf("Enter string %d: ", i+1);
        gets(strArr[i]);
    }

    insertionSort(strArr, n);
    printf("\nSorted array:\n");
    for(i = 0; i < n; i++) {
        printf("%s\n", strArr[i]);
    }

    return 0;
}
