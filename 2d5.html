                int i, j;

            void copy(int arr[2][2], int arr2[2][2], int rows, int cols)
            {
                int i, j;
            
                for (i = 0; i < rows; i++)
                {
            
                    for (j = 0; j < cols; j++)
                    {
                        arr2[i][j] = arr[i][j];
                    }
                }
            }
            int main()
            {
                int arr[2][2], i, j;
                int rows = 2, cols = 2;
                int arr2[2][2];
                assign(arr, rows, cols);

                printf("The copied matrix is:\n");
