        #include <stdio.h>
            int linear(char *arr, int num, char key)
            {
                if (num == 0)
                    return -1;
                else if (arr[num - 1] == key)
                    return num - 1;
                else
                    linear(arr, num - 1, key);
            }
            int main()
            {
                char arr[10], key;
                int n = 0, i = 0, k;
                printf("Enter the string:");
                gets(arr);
                while (arr[i] != '\0')
                {
                    n++;
                    i++;
                }
                printf("Enter the element you want to find:");
                scanf("%c", &key);
                k = linear(arr, n, key);
                if (k == -1)
                    printf("Element not found\n");
                else
                    printf("Found at index %d", k);
            
                return 0;
            }
